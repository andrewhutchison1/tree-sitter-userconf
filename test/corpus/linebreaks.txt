lb.can_sep_kv | Linebreaks can separate the key and value in a record
===
key
    value

key ; comment
    value
---
(document
    (record_entry
        key:    (unquoted_string)
        value:  (unquoted_string))
    (record_entry
        key:    (unquoted_string)
        (comment)
        value:  (unquoted_string)))

===
lb.can_sep | Linebreaks can act as separators in records and lists
===
"key"       "value"
"key"       "value"

"key"       "value"

"record"    {
    "key" "value"
    "key" "value"
}

"list" [
    "1"
    "2"


    "3"]
---
(document
    (record_entry
        key: (quoted_string)
        value: (quoted_string))
    (record_entry
        key: (quoted_string)
        value: (quoted_string))
    (record_entry
        key: (quoted_string)
        value: (quoted_string))
    (record_entry
        key: (quoted_string)
        value:
            (record
                (record_entry key: (quoted_string) value: (quoted_string))
                (record_entry key: (quoted_string) value: (quoted_string))))
    (record_entry
        key:    (quoted_string)
        value:  (list (quoted_string) (quoted_string) (quoted_string))))
